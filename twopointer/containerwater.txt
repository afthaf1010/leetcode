class Solution:
    def maxArea(self, height):
        i=0
        j=len(height)-1
        m=0
        while i<j:
            w = j-i
            a=min(height[i], height[j]) * w
            m=max(m, a)
            if height[i]<height[j]:
                i+=1
            else:
                j-=1
        return m
